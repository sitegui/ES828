\section{Objetivo}
\begin{par}
	O objetivo deste pré-roteiro é encontrar os ganhos de um controlador que satisfaz os seguintes requisitos:
\end{par}


\begin{itemize}
 \item $\xi = \frac{\sqrt{2}}{2}$

 \item Terceiro polo alocado em $-30$

 \item Erro nulo para entrada do tipo rampa.
 
  \item Estabilizaçăo em $0,5 [s]$

\end{itemize}

\section{Localizaçăo dos polos}

\begin{par}
Com a especificaçăo do tempo de estabilizaçăo de $0,5[s]$ e com coeficiente de amortecimento de $\sqrt{2}/2$, pode-se encontrar $\omega_n$.
\end{par}

\begin {equation}
\xi = \sqrt{2}/2
\end {equation}

\begin {equation}
t_e = 4 / (\xi \cdot \omega_n)
\end {equation}

\begin {par}
Desta forma encontra-se $\omega_n=11.3137$. Ou seja, tem-se os seguintes polos dominantes:
\end{par}

\begin {equation}
p_1 = -8.0000 + 8.0000i
\end {equation}

\begin {equation}
p_2 = -8.0000 - 8.0000i
\end {equation}

\begin {par}
É necessário ainda definir o último polo, este foi definido arbitrariamente no ponto $p_3$=-30.
\end{par}

\section{Modelo de estados}

\begin {par}
A matriz que representa a planta no modelo de estados é definida como:
\end{par}


\[ \left( \begin{array}{ccc}
A & B \\
C & D \end{array} \right)\]

\begin {par}
Onde A, B, C e D săo representados por:
\end{par}

\[ A = \left( \begin{array}{ccc}
- 120 & -51.76 & 0 \\
64 & 0 & 0 \\
0 & 1 & 0 \end{array} \right).\] 

\[ B = \left( \begin{array}{c}
16\\
0 \\
0 \end{array} \right).\]

\[ C = \left( \begin{array}{ccc}
0 & 0 & 29.09  \end{array} \right).\]

\begin {equation}
D = (0)
\end {equation}

\section{Determinaçăo da matriz de ganho K}

\begin {par}
Com o auxílio do MATLAB pode-se encontrar o valor da matriz de ganho K via o seguinte comando:
\end{par}

\begin{verbatim}
K = place(A, B, p)
\end{verbatim}

\[ K = \left( \begin{array}{ccc}
-4.6242 & -2.6412 & 3.7500  \end{array} \right).\]

\section{Pólos do observador}

\begin {par}
Os pólos do observador devem ser escolhidos de forma que o estado estimado siga o estado verdadeiro o mais rápido possível. Para isso, eles devem estar mais ŕ esqueda que os pólos do controlador.
Dessa forma, foi escolhido arbitrariamente que a parte real deveria ser 50\% mais negativa.
\end{par}

\[ po = \left( \begin{array}{c}
-12.0000 + 8.0000i\\
-12.0000 - 8.0000i \\
-45.0000 + 0.0000i \end{array} \right).\]

\section{Pólos do observador}

\begin {par}
Com os pólos do observador calcula-se a matriz L, novamente via MATLAB com o seguinte comando:
\end{par}

\begin{verbatim}
L = place(A', C', po)'
\end{verbatim}

\[ B = \left( \begin{array}{c}
-168.0343\\
140.6948 \\
-1.7526 \end{array} \right).\]

\section{Determinaçăo de $M(s)$}

\begin{par}
Para encontrar o M(s), inicialmente encontra-se a funçăo $S(s)$ denominada por:
\end{par}

\begin{equation}
S = C*(s*I - (A - B*K))^{-1}*B
\end{equation}

\begin{equation}
S = \frac {2.979 *10^4} { s^3 + 46 s^2 + 608 s + 3840}
\end{equation}

\begin{par}
Encontrado a funçăo $S(s)$, o próximo passo é determinar $M(0)$
\end{par}

\begin{verbatim}
M0 = pinv(evalfr(S, 0)*K)
\end{verbatim}

\[ M(0) = \left( \begin{array}{c}
-0.0141\\
-0.0080 \\
0.0114 \end{array} \right).\]

\begin{par}
Supondo que a forma de $M(s)$ é:
\end{par}

\begin{equation}
M(s) = \frac{s/\tau + 1}{s/30 + 1} M0
\end{equation}

\begin{par}
Deve-se determinar $\tau$ de forma que:
\end{par}

\begin{equation}
(\frac{d}{ds} S(s) K M(s))_{s=0} = 0
\end{equation}

\begin{equation}
\tau = 5.2174
\end{equation}
