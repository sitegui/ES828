controlador 1

b=diff(1/G2);
sr=solve(b,x);
sr1=sr(1);

K=-1/subs(G2,x,sr1);
K=0.8926

%%resposta a rampa controlador 1
t=0:0.001:4;
rampa=t;
Y=minreal(G*K/(1+G*K))
step(Y/s, 4)
hold on
plot(rampa,t,'r'); title('Resposta à rampa do controlador proporcional');
